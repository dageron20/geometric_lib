def area(a, b): 
    return a * b 

def perimeter(a, b): 
    return (a + b)*2

import unittest

class rectangleTestCase(unittest.TestCase):
   def test_zero_mul(self):
       res = area(5, 0)
       self.assertEqual(res, 0)
       
   def test_square_mul(self):
       res = area(5, 5)
       self.assertEqual(res, 25)
